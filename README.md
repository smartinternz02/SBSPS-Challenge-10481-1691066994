def convert_image_to_array(image_dir):
  try:
    image = cv2.imread(image_dir)

      image = cv2.resize(image, (256,256))
      #image = cv2.cvtColor(image, cv2.COLOR_BGR2GRAY)
      return img_to_array(image)
    else:
      return np.array([])
  except Exception as e:
    print(f"Error : {e}")
    return None
